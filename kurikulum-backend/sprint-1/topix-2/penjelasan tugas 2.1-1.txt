var_dump($e == $f); : mengatahui apa veriabel true atau false dalam persaman
var_dump($e === $f); :mengatahui apa identik veriabel tersebut true atau false dalam data veriabel

var_dump((int)$g); : mengatahui apakah di $g tersebut ada bilangan desimal
